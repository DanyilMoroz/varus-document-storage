1. Course table:
	create table course(
	id bigint identity not null, 
	name varchar not null,
	coursecategory_id bigint,
	foreign key(coursecategory_id) references course_category(id)
	);

2. Insert Course category:
	insert into course_category(id, name) values(default, 'IT');
	insert into course_category(id, name) values(default, 'Art & Design');
	insert into course_category(id, name) values(default, 'Data Science');
	insert into course_category(id, name) values(default, 'Mathematics');
	insert into course_category(id, name) values(default, 'Science');
	insert into course_category(id, name) values(default, 'Programming');

3. Insert Course:
	insert into course(id, name, coursecategory_id) values (default, 'Java Programming', 1);

4. Select Courses:
	select course.id as courseId, course.name as courseName, course_category.id as CategoryId, 
	course_category.name as categoryName from course left join course_category on course.coursecategory_id = course_category.id;
